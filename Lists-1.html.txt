<script>
// Supabase configuration
const SUPABASE_URL = 'https://raaylemyoajafvufkldf.supabase.co';
const SUPABASE_ANON_KEY = 'eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJpc3MiOiJzdXBhYmFzZSIsInJlZiI6InJhYXlsZW15b2FqYWZ2dWZrbGRmIiwicm9sZSI6ImFub24iLCJpYXQiOjE3NjcwMjE0MDAsImV4cCI6MjA4MjU5NzQwMH0.8qHi4IPItPZEXAzD3nL_M5PBDfADAk-XntODwYZyaG8';

// Initialize Supabase client
const supabase = createClient(SUPABASE_URL, SUPABASE_ANON_KEY);

// Global variables
let currentPage = 0;
const articlesPerPage = 10;
let allArticles = [];
let filteredArticles = [];
let isLoading = false;

// Initialize the app
document.addEventListener('DOMContentLoaded', function() {
  loadTheme();
  loadArticles();
  setupSearch();
});

// Theme management
function toggleTheme() {
  document.body.classList.toggle('light-theme');
  localStorage.setItem("theme", document.body.classList.contains("light-theme") ? "light" : "dark");
}

function loadTheme() {
  if (localStorage.getItem("theme") === "light") {
    document.body.classList.add("light-theme");
  }
}

// Search functionality
function setupSearch() {
  const searchInput = document.getElementById('searchInput');
  let searchTimeout;
  
  searchInput.addEventListener('input', function() {
    clearTimeout(searchTimeout);
    searchTimeout = setTimeout(() => {
      const query = this.value.trim();
      if (query) {
        searchArticles(query);
      } else {
        clearSearch();
      }
    }, 300);
  });
  
  searchInput.addEventListener('keypress', function(e) {
    if (e.key === 'Enter') {
      e.preventDefault();
      const query = this.value.trim();
      if (query) {
        searchArticles(query);
      } else {
        clearSearch();
      }
    }
  });
}

function searchArticles(query = null) {
  const searchQuery = query || document.getElementById('searchInput').value.trim();
  if (!searchQuery) {
    clearSearch();
    return;
  }
  
  filteredArticles = allArticles.filter(article => 
    article.title.toLowerCase().includes(searchQuery.toLowerCase()) ||
    article.summary.toLowerCase().includes(searchQuery.toLowerCase()) ||
    (article.users && article.users.full_name.toLowerCase().includes(searchQuery.toLowerCase()))
  );
  
  currentPage = 0;
  displayArticles();
}

function clearSearch() {
  document.getElementById('searchInput').value = '';
  filteredArticles = [...allArticles];
  currentPage = 0;
  displayArticles();
}

// Article loading and display
async function loadArticles() {
  if (isLoading) return;
  
  isLoading = true;
  showLoadingState();
  
  try {
    const { data, error } = await supabase
      .from('articles')
      .select(`
        *,
        users!inner(id, full_name, display_name, profile_photo_url),
        article_stats!inner(views_count, likes_count, comments_count)
      `)
      .eq('status', 'published')
      .order('published_at', { ascending: false })
      .limit(50); // Load more articles initially
    
    if (error) throw error;
    
    allArticles = data || [];
    filteredArticles = [...allArticles];
    currentPage = 0;
    
    hideLoadingState();
    displayArticles();
    
  } catch (error) {
    console.error('Error loading articles:', error);
    showErrorState();
  } finally {
    isLoading = false;
  }
}

function displayArticles() {
  const articleList = document.getElementById('articleList');
  const loadMoreContainer = document.getElementById('loadMoreContainer');
  
  if (filteredArticles.length === 0) {
    articleList.innerHTML = `
      <div class="error-state">
        <i class="fas fa-search" style="font-size: 24px; margin-bottom: 16px;"></i>
        <p>No articles found.</p>
      </div>
    `;
    articleList.style.display = 'block';
    loadMoreContainer.style.display = 'none';
    return;
  }
  
  const startIndex = 0;
  const endIndex = (currentPage + 1) * articlesPerPage;
  const articlesToShow = filteredArticles.slice(startIndex, endIndex);
  
  articleList.innerHTML = articlesToShow.map((article, index) => {
    const authorName = article.users?.display_name || article.users?.full_name || 'Unknown Author';
    const readTime = article.read_time_minutes || Math.ceil((article.summary || '').length / 200);
    const publishedDate = new Date(article.published_at || article.created_at).toLocaleDateString('en-US', {
      year: 'numeric',
      month: 'short',
      day: 'numeric'
    });
    
    return `
      <li class="article-item" onclick="openArticle('${article.id}', '${article.slug}')" style="animation-delay: ${index * 0.05}s">
        <h2 class="article-title">${escapeHtml(article.title)}</h2>
        <div class="article-meta">
          <span class="author-name">${escapeHtml(authorName)}</span>
          <div class="meta-dot"></div>
          <span>${publishedDate}</span>
          <div class="meta-dot"></div>
          <span>${readTime} min read</span>
          ${article.category ? `<span class="category-tag">${escapeHtml(article.category)}</span>` : ''}
        </div>
      </li>
    `;
  }).join('');
  
  articleList.style.display = 'block';
  
  // Show/hide load more button
  if (endIndex < filteredArticles.length) {
    loadMoreContainer.style.display = 'block';
  } else {
    loadMoreContainer.style.display = 'none';
  }
}

function loadMoreArticles() {
  currentPage++;
  displayArticles();
}

// Utility functions
function escapeHtml(text) {
  const div = document.createElement('div');
  div.textContent = text;
  return div.innerHTML;
}

function openArticle(articleId, articleSlug) {
  window.location.href = `Articls.html?id=${articleId}&slug=${articleSlug}`;
}

function goHome() {
  window.location.href = 'Lists.html';
}

function goToProfile() {
  // For now, go to a default user profile
  // In a real app, this would use the logged-in user's ID
  window.location.href = 'User.html?id=test-user';
}

// State management functions
function showLoadingState() {
  document.getElementById('loadingState').style.display = 'block';
  document.getElementById('errorState').style.display = 'none';
  document.getElementById('articleList').style.display = 'none';
  document.getElementById('loadMoreContainer').style.display = 'none';
}

function hideLoadingState() {
  document.getElementById('loadingState').style.display = 'none';
}

function showErrorState() {
  document.getElementById('loadingState').style.display = 'none';
  document.getElementById('errorState').style.display = 'block';
  document.getElementById('articleList').style.display = 'none';
  document.getElementById('loadMoreContainer').style.display = 'none';
}
